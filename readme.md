<h2>Install</h2>

<ol>
	<li>git clone https://github.com/t28708/laravel.git [name folder]</li>
    <li>cd [name folder]</li>
	<li>php artisan config:cache</li>
	<li>artisan storage:link</li>
	<li>artisan migrate</li>
	<li>npm install</li>
	<li>npm audit fix</li>
	<li>npm run dev</li>
</ol>
